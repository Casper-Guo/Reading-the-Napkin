\chapter{Groups}

\section{Definition and examples of groups}

\begin{definition}
  A group is a pair consisting of a set $G$ and a binary operation $\star$ on $G$ such that:
  \begin{itemize}
    \item $G$ has an identity element
    \item The operation $\star$ is associative
    \item Every element $g \in G$ has an inverse
  \end{itemize}
  Closure of the set $G$ under operation $\star$ is implied.
\end{definition}

It is not required that $\star$ is commutative. The group is \textbf{abelian} if the binary operation is commutative and \textbf{non-abelian} otherwise.

\begin{definition}
  The cyclic group of order $n$ is the group under addition of the remainders modulo $n$ where $n \in \Z \; n > 1$.
\end{definition}

\begin{notation}
  The cyclic group of order $n$ is denoted $\Zc n$.
\end{notation}

\begin{notation}
  Let $p$ be a prime, denote the nonzero residues modulo $p$ as $(\Zc p)^\times$
\end{notation}

\section{Isomorphisms}
\begin{definition}
  A bijection $\phi \colon G \to H$ is called an isomorphism if:

  \[ \phi(g_1 \star g_2) = \phi(g_1) \ast \phi(g_2) \quad \text{for all $g_1, g_2 \in G$} \]
\end{definition}

\begin{solution}
  Exercise 1.3.5: Let the primitive root modulo $p$ be $r$. We use the same bijection $\phi(a \mod p - 1) = r^a \mod p$. Notice that this map is a bijection by the definition of a primitive root. Then we check $\phi(a \star b) = \phi(a) \ast \phi(b)$, which is $p^{a + b \mod p -1} \equiv p^{a \mod p - 1}p^{b \mod p - 1} \mod p$. Thus the two groups are isomorphic.
\end{solution}

$\cong$ (being isomoprhic) is an equivalence relation because:

\begin{itemize}
  \item Reflexive: Any group is isomorphic to itself via the identity map
  \item Symmetric: An isomorphism is a bijection, so there must be a reverse map
  \item Transitive: Since isomorphisms are just bijective functions, they can be composed
\end{itemize}

\section{Orders of groups, and Lagrange's theorem}
\begin{solution}
  Question 1.4.3: For the group $\Zc 6$,
  \begin{itemize}
    \item $\ord 0 = 1$
    \item $\ord 1 = 6$
    \item $\ord 2 = 3$
    \item $\ord 3 = 2$
    \item $\ord 4 = 3$
    \item $\ord 5 = 6$
  \end{itemize}
\end{solution}

\begin{proof}
  Consider an element $g \in G$ and the sequence $x^0 = 1_G, g, g^2, \cdots, g^{\abs{G}}$ of length $\abs{G} + 1$. By the pigeonhole principle, two elements of the sequence must be equal, call them $g^x$ and $g^y$. WOLOG assume $x < y$, then it must be the case that $g^{y - x} = 1_G$ and $\ord g = y -x$. Since $y - x \leq \abs{G}$, and $G$ is a finite group, $\ord g$ is finite as well.
\end{proof}

\begin{theorem}
  Lagrange's theorem for orders: Let $G$ be a finite group, then $x^{\abs{G}} = 1_g$ for all $x \in G$.
\end{theorem}

\section{Subgroups}
\begin{definition}
  $\GL_n(\R)$, the general linear group, is the group of $n \times n$ matrices with non-zero determinants under matrix multiplication. The special linear group, $\SL_n(\R)$, is the group of $n \times n$ matrices with determinant 1 under matrix multiplication.
\end{definition}

$\SL_n(\R)$ is a subgroup of $\GL_n(\R)$.

\begin{definition}
  The subgroup generated by $x \in G$ is the set $\set{\cdots, x^{-1}, 1_G, x, \cdots}$.
\end{definition}

\begin{solution}
  Exercise 1.5.6: If $\ord x = 2015$, the subgroup generated by $x$ is the set $\set{1_G, x, \cdots, x^{2014}}$. If $\ord x = \infty$, the subgroup generated by $x$ is the set $\set{\cdots, x^{-1}, 1_G, x, \cdots}$.
\end{solution}

\section{Groups of small orders}
$D_6 \cong S_3$ because the transformations of an equilateral triangle correspond to the permutation of its vertices.

$\Zc 6 \cong \Zc 2 \times \Zc 3$ and $\Zc{10} \cong \Zc 2 \times \Zc 5$ because of the Chinese remainder theorem. Given that 2 is coprime with 3 and 5, a number can be uniquely identified by its remainder modulo 2 and 3 up to the lowest common multiple which is 6. Same for modulo 2 and 5.
